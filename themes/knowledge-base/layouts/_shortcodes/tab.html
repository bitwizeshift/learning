{{- /* A single tab within {{< tabs >}}. Accepts `icon` (a named SVG icon) and/or `label` (text; also positional 0). Either or both may be set. {{< tab icon="python" label="Python" >}}...{{< /tab >}} {{< tab "Plain text tab" >}}...{{< /tab >}} Group identity and the 1-based index are derived from the parent instance so IDs stay unique without depending on template execution order. The page's UniqueID is folded in too, so summaries rendered together on a list page (each with its own ordinal counter) never share a radio group `name`. */ -}} {{- $uid := "" -}} {{- with .Page.File }}{{ $uid = .UniqueID }}{{ end -}} {{- $gid := printf "%s-%d" $uid .Parent.Ordinal -}} {{- $idx := add (.Parent.Store.Get "count" | default 0) 1 -}} {{- .Parent.Store.Set "count" $idx -}} {{- $name := printf "tabset-%s" $gid -}} {{- $id := printf "tab-%s-%d" $gid $idx -}} {{- $icon := .Get "icon" -}} {{- $label := .Get "label" | default (.Get 0) -}}
